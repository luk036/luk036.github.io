[docs]class dllink:
    """doubly linked node (that may also be a "head" a list)

    A Doubly-linked List class. This class simply contains a link of
    node's. By adding a "head" node (sentinel), deleting a node is
    extremely fast (see "Introduction to Algorithm"). This class does
    not keep the length information as it is not necessary for the FM
    algorithm. This saves memory and run-time to update the length
    information. Note that this class does not own the list node. They
    are supplied by the caller in order to better reuse the nodes.
    """

    def __init__(self, idx=None):
        """initialization

        Keyword Arguments:
            idx (type):  description (default: {None})
        """
        self.next = self.prev = self
        self.idx = idx
        self.key = 0


[docs]    def detach(self):
        """detach from a list """
        assert self.next is not None
        n = self.next
        p = self.prev
        p.next = n
        n.prev = p



[docs]    def lock(self):
        """lock the node (and don't append it to any list) """
        self.next = None

        """whether the node is locked

        Returns:
            bool:  description
        """
        return self.next is None

        """whether the list is empty

        Returns:
            bool:  description
        """
        return self.next == self



[docs]    def clear(self):
        """clear"""
        self.next = self.prev = self



[docs]    def appendleft(self, node):
        """append the node to the front

        Arguments:
            node {dllink}:  description
        """
        node.next = self.next
        self.next.prev = node
        self.next = node
        node.prev = self



[docs]    def append(self, node):
        """append the node to the back

        Arguments:
            node {dllink}:  description
        """
        node.prev = self.prev
        self.prev.next = node
        self.prev = node
        node.next = self



[docs]    def popleft(self):
        """pop a node from the front

        Returns:
            dllink:  description
        """
        res = self.next
        self.next = res.next
        self.next.prev = self
        return res



[docs]    def pop(self):
        """pop a node from the back

        Returns:
            dllink:  description
        """
        res = self.prev
        self.prev = res.prev
        self.prev.next = self
        return res


    def __iter__(self):
        """iterable

        Returns:
            dllink:  itself
        """
        cur = self.next
        while cur != self:
            yield cur
            cur = cur.next
