    """Round Robin

    Raises:
        StopIteration:  description

    Returns:
        dtype:  description
    """

    def __init__(self, K):
        self.cycle = list(dllink(k) for k in range(K))
        K -= 1
        for k in range(K):
            self.cycle[k].next = self.cycle[k+1]
        self.cycle[K].next = self.cycle[0]
