31 class my_fir_oracle
32 {
33  using Cut = std::tuple<Arr, double>;
34 
35  // Gaussian filter with linear phase. (Uncomment lines below for this
36  // design.) var = 0.05 Hdes = 1/(xt::sqrt(2*PI*var)) *
37  // xt::exp(-xt::square(w-PI/2)/(2*var)) Hdes = xt::multiply(Hdes,
38  // xt::exp(-1j*n/2*w))
39  Arr w = xt::linspace<double>(0, PI, m);
40 
41  // A is the matrix used to compute the frequency response
42  // from a vector of filter coefficients:
43  // A[w,:] = [1 exp(-j*w) exp(-j*2*w) ... exp(-j*n*w)]
44  // CArr A = xt::exp(xt::linalg::kron(w, xt::arange(n)));
45 
46  // Presently CVXPY does not do complex-valued math, so the
47  // problem must be formatted into a real-valued representation.
48 
49  // Split Hdes into a real part, and an imaginary part.
50  // Arr Hdes_r = xt::real(Hdes);
51  // Arr Hdes_i = xt::imag(Hdes);
52  Arr Hdes_theta = D * w;
53  Arr Hdes_r = xt::cos(Hdes_theta);
54  Arr Hdes_i = -xt::sin(Hdes_theta);
55  Arr A_theta = xt::linalg::outer(w, xt::arange(n));
56 
57  // Split A into a real part, and an imaginary part.
58  // Arr A_R = xt::real(A);
59  // Arr A_I = xt::imag(A);
60  Arr A_R = xt::cos(A_theta);
61  Arr A_I = -xt::sin(A_theta);
62 
63  public:
64  auto operator()(const Arr& h, double t) const -> std::tuple<Cut, double>
65  {
66  auto fmax = std::numeric_limits<double>::min();
67  auto gmax = zeros({n});
68 
69  for (auto i = 0U; i < m; ++i)
70  {
71  auto a_R = Arr {xt::view(A_R, i, xt::all())};
72  auto a_I = Arr {xt::view(A_I, i, xt::all())};
73  auto H_r = Hdes_r[i];
74  auto H_i = Hdes_i[i];
75  auto t_r = xt::linalg::dot(a_R, h)() - H_r;
76  auto t_i = xt::linalg::dot(a_I, h)() - H_i;
77  auto fj = t_r * t_r + t_i * t_i;
78  if (fj >= t)
79  {
80  auto g = Arr {2. * (t_r * a_R + t_i * a_I)};
81  return {{std::move(g), fj - t}, t};
82  }
83  if (fmax < fj)
84  {
85  fmax = fj;
86  // imax = i;
87  gmax = 2. * (t_r * a_R + t_i * a_I);
88  }
89  }
90 
91  return {{std::move(gmax), 0.}, fmax};
92  }
93 };
