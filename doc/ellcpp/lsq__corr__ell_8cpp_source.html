40  * @brief Create a 2d isotropic object
41  *
42  * @param s location of sites
43  * @param N
44  * @return Arr Biased covariance matrix
45  */
46 Arr create_2d_isotropic(const Arr& s, size_t N = 3000U)
47 {
48  using xt::linalg::dot;
49 
50  const auto n = s.shape()[0];
51  const auto sdkern = 0.3; // width of kernel
52  const auto var = 2.; // standard derivation
53  const auto tau = 0.00001; // standard derivation of white noise
54  xt::random::seed(5);
55 
56  auto Sig = zeros({n, n});
57  for (auto i = 0U; i < n; ++i)
58  {
59  for (auto j = i; j < n; ++j)
60  {
61  auto d = xt::view(s, j, xt::all()) - xt::view(s, i, xt::all());
62  auto g = -sdkern * std::sqrt(dot(d, d)());
63  Sig(i, j) = std::exp(g);
64  Sig(j, i) = Sig(i, j);
65  }
66  }
67 
68  auto A = xt::linalg::cholesky(Sig);
69  auto Y = zeros({n, n});
70  for (size_t k = 0U; k < N; ++k)
71  {
72  auto x = var * xt::random::randn<double>({n});
73  auto y = dot(A, x) + tau * xt::random::randn<double>({n});
74  // Arr y = dot(A, x);
75  Y += xt::linalg::outer(y, y);
76  }
77  Y /= N;
78 
79  return Y;
80 }

97  auto d = std::sqrt(xt::linalg::dot(h, h)());

99  D1(j, i) = d;
100  }

178  g(n - 1) = 0.;
179  return {{std::move(g), f0}, t};

182

184  {
185  auto [g1, f1] = *cut1;
186  xt::view(g, xt::range(0, n - 1)) = g1;
187  auto& Q = this->_qmi._Q;
188  // auto ep = Q.witness();
189  auto [start, stop] = Q.p;
190  auto v = xt::view(Q.v, xt::range(start, stop));
191  g(n - 1) = -xt::linalg::dot(v, v)();
192  return {{std::move(g), f1}, t};
193  }
194  g(n - 1) = 1.;
195 
196  auto f0 = x(n - 1) - t;
197  if (f0 > 0)
198  {
199  return {{std::move(g), f0}, t};
200  }
201  return {{std::move(g), 0.}, x(n - 1)};
202  }

301 
302  const auto& R = this->_lmi0._Q.sqrt();
303  auto invR = Arr {xt::linalg::inv(R)};

305  auto SY = Arr {dot(S, this->_Y)};
306 
307  auto diag = xt::diagonal(R);
308  auto f1 =
309  double {2. * xt::sum(xt::log(diag))() + xt::linalg::trace(SY)()};
310  // auto f1 = 0.;
311 
312  auto f = f1 - t;
313  if (f < 0)
314  {
315  t = f1;
316  f = 0.;
317  }
318 
319  auto g = zeros(x);
320 
321  for (size_t i = 0U; i < n; ++i)
322  {
323  auto SFsi = dot(S, this->_Sig[i]);
324  g(i) = xt::linalg::trace(SFsi)();
325  for (size_t k = 0U; k < m; ++k)
326  {
327  g(i) -= dot(
