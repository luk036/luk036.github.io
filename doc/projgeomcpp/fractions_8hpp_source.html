35  * @brief Least common multiple
36  *
37  * @tparam _Mn
38  * @param __m
39  * @param __n
40  * @return _Mn
41  */
42 template<Integral _Mn>
43 constexpr _Mn lcm(_Mn __m, _Mn __n)
44 {
45  return (__m != 0 && __n != 0) ? (abs(__m) / gcd(__m, __n)) * abs(__n) : 0;
