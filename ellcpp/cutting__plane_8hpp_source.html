43 auto bsearch(Oracle &Omega, Space &I, const Options &options = Options()) {
44  // assume monotone
45  auto &[l, u] = I;
46  auto u_orig = u;
47  auto niter = 1U;
48  for (; niter <= options.max_it; ++niter) {
49  auto t = l + (u - l) / 2;
50  if (Omega(t)) { // feasible sol'n obtained
51  u = t;
52  } else {
53  l = t;
54  }
55  auto tau = (u - l) / 2;
56  if (tau < options.tol) {
57  break;
58  }
59  }
60  auto feasible = (u != u_orig);
61  auto ret = CInfo(feasible, niter, 0);
62  ret.value = u;
63  return std::move(ret);
64 }
