45 auto bsearch(Oracle& Omega, Space& I, const Options& options = Options())
46 {
47  // assume monotone
48  auto& [l, u] = I;
49  auto u_orig = u;
50  auto niter = 1U;
51  for (; niter <= options.max_it; ++niter)
52  {
53  auto t = l + (u - l) / 2;
54  if (Omega(t))
55  { // feasible sol'n obtained
56  u = t;
57  }
58  else
59  {
60  l = t;
61  }
62  auto tau = (u - l) / 2;
63  if (tau < options.tol) { break; }
64  }
65  auto feasible = (u != u_orig);
66  auto ret = CInfo(feasible, niter, 0);
67 
68  ret.value = u;
69  return std::move(ret);
