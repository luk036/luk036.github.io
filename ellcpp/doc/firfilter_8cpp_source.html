38  // Gaussian filter with linear phase. (Uncomment lines below for this
39  // design.) var = 0.05 Hdes = 1/(xt::sqrt(2*PI*var)) *
40  // xt::exp(-xt::square(w-PI/2)/(2*var)) Hdes = xt::multiply(Hdes,
41  // xt::exp(-1j*n/2*w))
42  Arr w = xt::linspace<double>(0, PI, m);
43 
44  // A is the matrix used to compute the frequency response
45  // from a vector of filter coefficients:
46  // A[w,:] = [1 exp(-j*w) exp(-j*2*w) ... exp(-j*n*w)]
47  // CArr A = xt::exp(xt::linalg::kron(w, xt::arange(n)));
48 
49  // Presently CVXPY does not do complex-valued math, so the
50  // problem must be formatted into a real-valued representation.
51 
52  // Split Hdes into a real part, and an imaginary part.
53  // Arr Hdes_r = xt::real(Hdes);
54  // Arr Hdes_i = xt::imag(Hdes);
55  Arr Hdes_theta = D * w;
56  Arr Hdes_r = xt::cos(Hdes_theta);
57  Arr Hdes_i = -xt::sin(Hdes_theta);
58  Arr A_theta = xt::linalg::outer(w, xt::arange(n));
59 
60  // Split A into a real part, and an imaginary part.
61  // Arr A_R = xt::real(A);

63  Arr A_R = xt::cos(A_theta);
64  Arr A_I = -xt::sin(A_theta);

66 public:
67  auto operator()(const Arr& h, double t) const -> std::tuple<Arr, double, double>
68  {
69  auto fmax = std::numeric_limits<double>::min();
70  auto gmax = Arr{xt::zeros<double>({n})};
71 
72  for (auto i = 0U; i < m; ++i)
73  {
74  auto a_R = Arr{xt::view(A_R, i, xt::all())};
75  auto a_I = Arr{xt::view(A_I, i, xt::all())};
76  auto H_r = Hdes_r[i];
77  auto H_i = Hdes_i[i];
78  auto t_r = xt::linalg::dot(a_R, h)() - H_r;
79  auto t_i = xt::linalg::dot(a_I, h)() - H_i;
80  auto fj = t_r * t_r + t_i * t_i;
81  if (fj >= t)
82  {
83  auto g = Arr{2. * (t_r * a_R + t_i * a_I)};
84  return std::tuple{std::move(g), fj - t, t};
85  }
86  if (fmax < fj)
87  {
88  fmax = fj;
89  // imax = i;
90  gmax = 2. * (t_r * a_R + t_i * a_I);
91  }
92  }
93 
94  return std::tuple{std::move(gmax), 0., fmax};
95  }
