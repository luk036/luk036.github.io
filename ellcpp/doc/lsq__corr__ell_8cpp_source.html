27 std::tuple<Arr, Arr> create_2d_isotropic(size_t nx = 10u, size_t ny = 8u,
28  size_t N = 3000u) {
29  using xt::linalg::dot;
30 
31  const auto n = nx * ny;
32  const auto s_end = Arr{10., 8.};
33  const auto sdkern = 0.3; // width of kernel
34  const auto var = 2.; // standard derivation
35  const auto tau = 0.00001; // standard derivation of white noise
36  xt::random::seed(5);
37 
38  // create sites s
39  auto sx = xt::linspace<double>(0., s_end[0], nx);
40  auto sy = xt::linspace<double>(0., s_end[1], ny);
41  auto [xx, yy] = xt::meshgrid(sx, sy);
42  auto st = Arr{xt::stack(xt::xtuple(xt::flatten(xx), xt::flatten(yy)), 0)};
43  auto s = xt::transpose(st);
44 
45  auto Sig = Arr{xt::zeros<double>({n, n})};
46  for (auto i = 0U; i < n; ++i) {
47  for (auto j = i; j < n; ++j) {
48  auto d = xt::view(s, j, xt::all()) - xt::view(s, i, xt::all());
49  auto g = -sdkern * std::sqrt(dot(d, d)());
50  Sig(i, j) = std::exp(g);
51  Sig(j, i) = Sig(i, j);
52  }
53  }
54 
55  auto A = xt::linalg::cholesky(Sig);
56  auto Y = Arr{xt::zeros<double>({n, n})};
57  for (auto k = 0U; k < N; ++k) {
58  auto x = var * xt::random::randn<double>({n});
59  auto y = dot(A, x) + tau * xt::random::randn<double>({n});
60  // Arr y = dot(A, x);
61  Y += xt::linalg::outer(y, y);
62  }
63  Y /= N;
64 
65  return std::tuple{std::move(Y), std::move(s)};
66 }

74 std::vector<Arr> construct_distance_matrix(const Arr &s, size_t m) {
75  auto n = s.shape()[0];
76  // c = cvx.Variable(m)
77  auto D1 = Arr{xt::zeros<double>({n, n})};
78  for (auto i = 0U; i < n; ++i) {
79  for (auto j = i + 1; j < n; ++j) {
80  auto h = xt::view(s, j, xt::all()) - xt::view(s, i, xt::all());
81  auto d = std::sqrt(xt::linalg::dot(h, h)());
82  D1(i, j) = d;
83  D1(j, i) = d;
84  }
85  }
86 
87  auto D = Arr{xt::ones<double>({n, n})};
88  auto Sig = std::vector{ D };
89  Sig.reserve(m);
90 
91  for (auto i = 0U; i < m - 1; ++i) {
92  D *= D1;
93  Sig.push_back(D);
94  }
95  return std::move(Sig);
96 }

234  auto operator()(const Arr &x, double t) {
235  using xt::linalg::dot;
236 
237  auto [g1, fj1, feasible1] = this->_lmi(x);
238  if (!feasible1) {
239  return std::tuple{std::move(g1), fj1, t};
240  }
241 
242  auto [g0, fj0, feasible0] = this->_lmi0(x);
243  if (!feasible0) {
244  return std::tuple{std::move(g0), fj0, t};
245  }
246 
247  auto n = x.shape()[0];
248  auto m = this->_Y.shape()[0];
249 
250  const auto &R = this->_lmi0._Q.sqrt();
251  auto invR = Arr{xt::linalg::inv(R)};
252  auto S = Arr{dot(invR, xt::transpose(invR))};
253  auto SY = Arr{dot(S, this->_Y)};
254 
255  auto diag = xt::diagonal(R);
256  auto f1 =
257  double{2. * xt::sum(xt::log(diag))() + xt::linalg::trace(SY)()};
258  // auto f1 = 0.;
259 
260  auto f = f1 - t;
261  if (f < 0) {
262  t = f1;
263  f = 0.;
264  }
265 
266  Arr g = xt::zeros<double>({n});
267 
268  for (auto i = 0U; i < n; ++i) {
269  auto SFsi = dot(S, this->_Sig[i]);
270  g(i) = xt::linalg::trace(SFsi)();
271  for (auto k = 0U; k < m; ++k) {
272  g(i) -= dot(xt::view(SFsi, k, xt::all()),
273  xt::view(SY, xt::all(), k))();
274  }
275  }
276  return std::tuple{std::move(g), f, t};
