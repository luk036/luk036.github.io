39

41  auto&& [l1, l2, l3] = trilateral;
42 
43  // using P = decltype(a1);
44  using L = decltype(l1);
45  using K = Value_type<P>;
46  static_assert(Projective_plane_prim2<L>);
47  static_assert(CommutativeRing<K>);
48 
49  auto zero = std::array<K, 3>{0, 0, 0};
50 
51  auto [t1, t2, t3] = tri_altitude(triangle);
52 
53  auto t4 = harm_conj(t1, t2, t3);
54  auto o = orthocenter(triangle);
55  auto tau = reflect(l1);
56  auto Q = tri_quadrance(triangle);
57  auto S = tri_spread(trilateral);
58 
59  auto [m12, m23, m13] = tri_midpoint(triangle);
60 
61  auto mt1 = a1 * m23;
62  auto mt2 = a2 * m13;
63  auto mt3 = a3 * m12;
64 
65  auto&& [q1, q2, q3] = Q;
66  auto&& [s1, s2, s3] = S;
67 
68  auto tqf = sq(q1 + q2 + q3) - 2 * (q1 * q1 + q2 * q2 + q3 * q3);
69  auto tsf = sq(s1 + s2 + s3) - 2 * (s1 * s1 + s2 * s2 + s3 * s3) - 4 * s1 * s2 * s3;
70  auto c3 = sq(q1 + q2 - q3) / (4 * q1 * q2);
71 
72  auto a3p = plucker(3, a1, 4, a2);
73  auto q1p = quadrance(a2, a3p);
74  auto q2p = quadrance(a1, a3p);
75  auto q3p = quadrance(a1, a2);
76  auto tqf2 = Ar(q1p, q2p, q3p); // get 0
77 
78  if constexpr (Integral<K>)
79  {
80  CHECK(!is_parallel(l1, l2));
81  CHECK(!is_parallel(l2, l3));
82  CHECK(is_perpendicular(t1, l1));
83  CHECK(spread(t1, l1) == 1);
84  CHECK(coincident(t1, t2, t3));
85  CHECK(R(t1, t2, t3, t4) == -1);
86  CHECK(o == t2 * t3);
87  CHECK(tau(tau(a1)) == a1);
88  CHECK(spread(l1, l1) == 0);
89  CHECK(quadrance(a1, a1) == 0);
90  CHECK(check_sine_law(Q, S));
91  CHECK(check_sine_law(S, Q));
92  CHECK(coincident(mt1, mt2, mt3));
93  // CHECK(cross_s(l1, l2) == c3);
94  CHECK((c3 + s3) == 1);
95  CHECK(tqf == Ar(q1, q2, q3));
96  CHECK(tsf == 0);
97  CHECK(tqf2 == 0);
98  CHECK(a1 == orthocenter(std::tuple{std::move(o), std::move(a2), std::move(a3)}));
99  }
100  else

103  CHECK(cross2(l2, l3) != Zero);
104  CHECK(dot1(t1, l1) == Zero);

108  CHECK(ApproxEqual(cross(o, meet(t2, t3)), zero));
109  CHECK(ApproxEqual(cross(tau(tau(a1)), a1), zero));
110  CHECK(mt1.dot(mt2 * mt3) == Zero);

114  CHECK(distance(a1, a1) == Zero);

116  CHECK((c3 + s3) - 1 == Zero);
117  CHECK(tqf - Ar(q1, q2, q3) == Zero);
118  CHECK(tsf == Zero);
119  CHECK(tqf2 == Zero);

123  }
